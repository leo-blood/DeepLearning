## 阶段一：文档索引（ingest.py）

### 步骤 1：加载多种格式的文档

```python
# ingest.py
import os
from pathlib import Path
from langchain_community.document_loaders import (
    PyPDFLoader,
    TextLoader,
    UnstructuredMarkdownLoader,
)
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma

DOCS_DIR = "./docs"
CHROMA_DIR = "./chroma_db"


def load_documents(docs_dir: str):
    """加载目录下所有支持的文档"""
    documents = []
    
    for file_path in Path(docs_dir).rglob("*"):
        if file_path.suffix == ".pdf":
            loader = PyPDFLoader(str(file_path))
        elif file_path.suffix == ".md":
            loader = UnstructuredMarkdownLoader(str(file_path))
        elif file_path.suffix == ".txt":
            loader = TextLoader(str(file_path), encoding="utf-8")
        else:
            continue
        
        docs = loader.load()
        # 确保 metadata 中有文件名
        for doc in docs:
            doc.metadata["filename"] = file_path.name
        documents.extend(docs)
        print(f"加载：{file_path.name}（{len(docs)} 页/块）")
    
    return documents
```

### 步骤 2：智能分块

```python
def split_documents(documents):
    """分块，中文文档用"。"作为优先分割点"""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=800,
        chunk_overlap=150,
        length_function=len,
        separators=["\n\n", "\n", "。", "！", "？", ".", "!", "?", " ", ""],
    )
    
    chunks = splitter.split_documents(documents)
    print(f"文档 {len(documents)} 页 → 分块 {len(chunks)} 块")
    return chunks
```

### 步骤 3：向量化并存储

```python
def build_vectorstore(chunks):
    """构建向量库"""
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    
    # 增量更新：先检查是否已有数据
    if os.path.exists(CHROMA_DIR):
        vectorstore = Chroma(
            persist_directory=CHROMA_DIR,
            embedding_function=embeddings,
        )
        # 获取已有文件名，避免重复索引
        existing_sources = set(
            doc["filename"]
            for doc in vectorstore.get()["metadatas"]
            if doc
        )
        new_chunks = [
            chunk for chunk in chunks
            if chunk.metadata.get("filename") not in existing_sources
        ]
        if new_chunks:
            vectorstore.add_documents(new_chunks)
            print(f"新增 {len(new_chunks)} 块到向量库")
        else:
            print("没有新文档，向量库无需更新")
    else:
        vectorstore = Chroma.from_documents(
            documents=chunks,
            embedding=embeddings,
            persist_directory=CHROMA_DIR,
        )
        print(f"创建向量库，共 {len(chunks)} 块")
    
    return vectorstore


if __name__ == "__main__":
    docs = load_documents(DOCS_DIR)
    chunks = split_documents(docs)
    build_vectorstore(chunks)
    print("索引完成！")
```

## 阶段二：问答链（qa_chain.py）

```python
# qa_chain.py
from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_community.retrievers import BM25Retriever
from langchain.retrievers import EnsembleRetriever
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnableParallel, RunnablePassthrough

CHROMA_DIR = "./chroma_db"

RAG_PROMPT = ChatPromptTemplate.from_template("""
你是一个专业的文档问答助手。请根据以下上下文回答用户的问题。

规则：
1. 只基于提供的上下文回答，不要使用外部知识
2. 如果上下文中没有足够信息，明确说"根据现有文档，我无法回答这个问题"
3. 引用具体内容时，注明来自哪个文档

上下文：
{context}

问题：{question}

回答：""")


def build_retriever(vectorstore, all_chunks):
    """混合检索器：向量 + BM25"""
    vector_retriever = vectorstore.as_retriever(
        search_type="mmr",
        search_kwargs={"k": 5, "fetch_k": 20, "lambda_mult": 0.6},
    )
    
    bm25_retriever = BM25Retriever.from_documents(all_chunks)
    bm25_retriever.k = 5
    
    return EnsembleRetriever(
        retrievers=[bm25_retriever, vector_retriever],
        weights=[0.3, 0.7],
    )


def format_docs_with_source(docs) -> str:
    """格式化文档块，附带来源信息"""
    formatted = []
    for i, doc in enumerate(docs, 1):
        source = doc.metadata.get("filename", "未知")
        page = doc.metadata.get("page", "")
        source_info = f"{source}" + (f" 第{page+1}页" if page != "" else "")
        formatted.append(f"[来源 {i}: {source_info}]\n{doc.page_content}")
    return "\n\n".join(formatted)


def build_qa_chain(retriever):
    """构建带来源的 RAG chain"""
    llm = ChatOpenAI(model="gpt-4o", temperature=0, streaming=True)
    
    # 并行：同时获取文档和透传问题
    rag_chain = (
        RunnableParallel({
            "context_docs": retriever,
            "question": RunnablePassthrough(),
        })
        .assign(
            context=lambda x: format_docs_with_source(x["context_docs"]),
        )
        | RunnableParallel({
            "answer": RAG_PROMPT | llm | StrOutputParser(),
            "sources": lambda x: list({
                doc.metadata.get("filename", "未知")
                for doc in x["context_docs"]
            }),
        })
    )
    
    return rag_chain


# 全局初始化（服务启动时执行一次）
def init_qa_system():
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    vectorstore = Chroma(
        persist_directory=CHROMA_DIR,
        embedding_function=embeddings,
    )
    
    # 加载所有文档用于 BM25
    all_docs = vectorstore.get()
    from langchain_core.documents import Document
    all_chunks = [
        Document(page_content=content, metadata=meta)
        for content, meta in zip(all_docs["documents"], all_docs["metadatas"])
    ]
    
    retriever = build_retriever(vectorstore, all_chunks)
    chain = build_qa_chain(retriever)
    return chain
```

## 阶段三：FastAPI 接口（app.py）

```python
# app.py
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from qa_chain import init_qa_system
import asyncio
import json

app = FastAPI(title="文档问答系统")

# 启动时初始化
qa_chain = None

@app.on_event("startup")
async def startup():
    global qa_chain
    qa_chain = init_qa_system()
    print("问答系统初始化完成")


class QuestionRequest(BaseModel):
    question: str
    stream: bool = True


@app.post("/ask")
async def ask_question(request: QuestionRequest):
    if not qa_chain:
        raise HTTPException(status_code=503, detail="系统未初始化")
    
    if request.stream:
        # 流式输出
        async def generate():
            sources = []
            async for chunk in qa_chain.astream(request.question):
                if "answer" in chunk:
                    # 流式传输答案 token
                    yield f"data: {json.dumps({'type': 'token', 'content': chunk['answer']})}\n\n"
                elif "sources" in chunk:
                    sources = chunk["sources"]
            
            # 最后发送来源信息
            yield f"data: {json.dumps({'type': 'sources', 'content': sources})}\n\n"
            yield "data: [DONE]\n\n"
        
        return StreamingResponse(generate(), media_type="text/event-stream")
    
    else:
        # 非流式输出
        result = await qa_chain.ainvoke(request.question)
        return {
            "answer": result["answer"],
            "sources": result["sources"],
        }


@app.get("/health")
async def health():
    return {"status": "ok"}
```

## 阶段四：客户端测试

```python
# test_stream.py
import httpx
import json

def ask_with_stream(question: str):
    print(f"Q: {question}\nA: ", end="")
    
    with httpx.stream("POST", "http://localhost:8000/ask", json={
        "question": question,
        "stream": True,
    }, timeout=30) as response:
        for line in response.iter_lines():
            if line.startswith("data: "):
                data = line[6:]
                if data == "[DONE]":
                    break
                
                payload = json.loads(data)
                if payload["type"] == "token":
                    print(payload["content"], end="", flush=True)
                elif payload["type"] == "sources":
                    print(f"\n\n📄 来源：{', '.join(payload['content'])}")

if __name__ == "__main__":
    ask_with_stream("什么是分布式锁？它的实现原理是什么？")
    print()
    ask_with_stream("Squirrel 缓存的穿透问题如何解决？")
```

## 关键设计决策

| 决策点 | 选择 | 原因 |
|--------|------|------|
| 检索策略 | Ensemble（BM25+向量） | 关键词+语义双路，覆盖更全 |
| Embedding | text-embedding-3-small | 性价比最高，支持中文 |
| 向量库 | Chroma | 本地持久化，无需额外服务 |
| LLM | GPT-4o + streaming | 质量高，流式输出体验好 |
| 增量更新 | 按文件名去重 | 避免重复索引，节省费用 |
