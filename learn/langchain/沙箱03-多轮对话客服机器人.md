# 沙箱 03｜多轮对话客服机器人

> **目标**：用 LangGraph 构建一个有状态、可路由的客服机器人，支持多轮对话、工单创建、FAQ 查询和人工转接。

---

## 系统设计

```
┌────────────────────────────────────────────────────────┐
│                  客服机器人工作流                         │
│                                                        │
│  用户输入 ──► [意图分类] ──► FAQ / 工单 / 人工 / 闲聊    │
│                  ↓                                     │
│            ┌─────┴─────┐                              │
│         FAQ检索      工单处理      ←── 需要登录         │
│            │             │                             │
│         向量库         数据库                           │
│            │             │                             │
│            └──────┬──────┘                             │
│                   ↓                                    │
│           [整合回复生成] ──► 流式输出给用户              │
│                   ↓                                    │
│           [满意度评估] ──► 结束 or 转人工               │
└────────────────────────────────────────────────────────┘
```

## 阶段一：状态定义（state.py）

```python
# state.py
from typing import TypedDict, Annotated, List, Optional, Literal
from langgraph.graph.message import add_messages
from langchain_core.messages import BaseMessage

class CustomerServiceState(TypedDict):
    # 对话历史（add_messages reducer 追加而非替换）
    messages: Annotated[List[BaseMessage], add_messages]
    
    # 意图分类结果
    intent: Optional[Literal["faq", "ticket", "human", "chat", "unknown"]]
    
    # 用户信息
    user_id: Optional[str]
    is_authenticated: bool
    user_name: Optional[str]
    
    # 工单信息
    ticket_id: Optional[str]
    ticket_type: Optional[str]   # complaint/inquiry/feedback
    ticket_priority: Optional[Literal["low", "medium", "high", "urgent"]]
    
    # FAQ 检索结果
    faq_results: Optional[List[dict]]
    
    # 对话管理
    turn_count: int              # 已对话轮次
    escalate_to_human: bool      # 是否需要转人工
    satisfaction_score: Optional[int]  # 1-5 满意度
    
    # 错误信息
    error: Optional[str]
```

## 阶段二：FAQ 知识库（faq_kb.py）

```python
# faq_kb.py
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma
from langchain_core.documents import Document
from langchain_core.tools import tool
from typing import List

# 示例 FAQ 数据（实际从数据库加载）
FAQ_DATA = [
    {"q": "如何修改密码？", "a": "进入账户设置 → 安全 → 修改密码，需要验证手机号。", "tag": "账户"},
    {"q": "忘记密码怎么办？", "a": "点击登录页'忘记密码'，通过手机或邮箱验证后重置。", "tag": "账户"},
    {"q": "如何申请退款？", "a": "在订单详情页点击'申请退款'，7个工作日内处理。", "tag": "退款"},
    {"q": "退款需要多久？", "a": "退款申请通过后，3-7个工作日到账，具体视银行而定。", "tag": "退款"},
    {"q": "如何联系人工客服？", "a": "工作时间9:00-21:00，可通过在线聊天或400电话联系。", "tag": "联系"},
    {"q": "发票怎么开？", "a": "在订单页面申请发票，电子发票3个工作日内发送到邮箱。", "tag": "发票"},
    {"q": "如何修改收货地址？", "a": "订单发货前可在订单详情修改地址，发货后无法修改。", "tag": "物流"},
    {"q": "会员权益有哪些？", "a": "会员享有专属折扣、优先发货、专属客服通道等权益。", "tag": "会员"},
]

_faq_store = None

def get_faq_store() -> Chroma:
    global _faq_store
    if _faq_store is None:
        docs = [
            Document(
                page_content=f"问：{item['q']}\n答：{item['a']}",
                metadata={"question": item["q"], "answer": item["a"], "tag": item["tag"]}
            )
            for item in FAQ_DATA
        ]
        _faq_store = Chroma.from_documents(
            docs,
            OpenAIEmbeddings(model="text-embedding-3-small"),
            collection_name="faq",
        )
    return _faq_store


@tool
def search_faq(query: str) -> str:
    """在 FAQ 知识库中搜索常见问题答案。
    
    Args:
        query: 用户的问题
    """
    store = get_faq_store()
    results = store.similarity_search_with_score(query, k=3)
    
    if not results or results[0][1] > 0.8:  # 相似度阈值
        return "未找到相关FAQ，建议转人工或提交工单。"
    
    output = []
    for doc, score in results:
        if score <= 0.6:  # 只返回高置信度结果
            output.append(doc.metadata["answer"])
    
    return "\n".join(output) if output else "未找到匹配的FAQ。"
```

## 阶段三：节点逻辑（nodes.py）

```python
# nodes.py
from langchain_openai import ChatOpenAI
from langchain_core.messages import SystemMessage, AIMessage, HumanMessage
from langchain_core.prompts import ChatPromptTemplate
from state import CustomerServiceState
from faq_kb import search_faq
import json
import uuid

llm = ChatOpenAI(model="gpt-4o", temperature=0.3)
llm_fast = ChatOpenAI(model="gpt-4o-mini", temperature=0)


# ── 节点 1：意图分类 ──────────────────────────────────────

def classify_intent(state: CustomerServiceState) -> dict:
    """分类用户意图"""
    last_message = state["messages"][-1].content
    history = state["messages"][-6:-1]  # 最近5条历史
    
    response = llm_fast.invoke([
        SystemMessage("""分析用户消息的意图，返回 JSON：
{
  "intent": "faq|ticket|human|chat",
  "confidence": 0-1,
  "reason": "简短原因"
}

- faq：询问产品/服务相关问题（退款、密码、发票等）
- ticket：投诉、报障、需要人工处理的问题
- human：明确要求转人工
- chat：日常闲聊、打招呼"""),
        *history,
        HumanMessage(f"用户消息：{last_message}"),
    ])
    
    try:
        result = json.loads(response.content)
        return {
            "intent": result["intent"],
            "turn_count": state.get("turn_count", 0) + 1,
        }
    except json.JSONDecodeError:
        return {"intent": "unknown", "turn_count": state.get("turn_count", 0) + 1}


# ── 节点 2：FAQ 查询 ──────────────────────────────────────

def handle_faq(state: CustomerServiceState) -> dict:
    """处理 FAQ 类型问题"""
    user_question = state["messages"][-1].content
    
    # 调用 FAQ 工具
    faq_answer = search_faq.invoke({"query": user_question})
    
    # 用 LLM 将 FAQ 答案整合成自然对话
    response = llm.invoke([
        SystemMessage("""你是友好的客服助手。根据 FAQ 库的答案，
用自然、友好的语气回复用户。如果 FAQ 无法完全解答，说明情况并建议提交工单。"""),
        *state["messages"][-4:],
        HumanMessage(f"FAQ 库答案：{faq_answer}\n请据此回复用户。"),
    ])
    
    return {
        "messages": [AIMessage(response.content)],
        "faq_results": [{"answer": faq_answer}],
    }


# ── 节点 3：工单处理 ──────────────────────────────────────

def handle_ticket(state: CustomerServiceState) -> dict:
    """创建工单并告知用户"""
    if not state.get("is_authenticated"):
        return {
            "messages": [AIMessage(
                "提交工单需要先登录账户。请问您的账户手机号是多少？（我会核验身份）"
            )],
        }
    
    # 提取工单信息
    extraction = llm_fast.invoke([
        SystemMessage("""从对话中提取工单信息，返回 JSON：
{
  "type": "complaint|inquiry|feedback|bug",
  "priority": "low|medium|high|urgent",
  "summary": "一句话概括问题"
}"""),
        *state["messages"][-6:],
    ])
    
    try:
        ticket_info = json.loads(extraction.content)
    except json.JSONDecodeError:
        ticket_info = {"type": "inquiry", "priority": "medium", "summary": "用户问题"}
    
    # 模拟创建工单（实际接入工单系统）
    ticket_id = f"TK{uuid.uuid4().hex[:8].upper()}"
    
    reply = f"""已为您创建工单 **#{ticket_id}**

- **问题类型**：{ticket_info.get('type', '咨询')}
- **优先级**：{ticket_info.get('priority', 'medium')}
- **问题概述**：{ticket_info.get('summary', '待处理')}

我们的团队将在 **2小时内**（紧急问题30分钟内）与您联系。
您也可以通过工单号 #{ticket_id} 在"我的工单"页面查看进度。

还有其他需要帮助的吗？"""
    
    return {
        "messages": [AIMessage(reply)],
        "ticket_id": ticket_id,
        "ticket_type": ticket_info.get("type"),
        "ticket_priority": ticket_info.get("priority"),
    }


# ── 节点 4：闲聊处理 ──────────────────────────────────────

def handle_chat(state: CustomerServiceState) -> dict:
    """处理闲聊，引导回正题"""
    response = llm.invoke([
        SystemMessage("""你是客服助手，友好地回应用户，但适当引导到服务相关话题。
不要聊太多与产品无关的话题。"""),
        *state["messages"][-4:],
    ])
    return {"messages": [AIMessage(response.content)]}


# ── 节点 5：转人工 ────────────────────────────────────────

def escalate_to_human(state: CustomerServiceState) -> dict:
    """将对话转接人工客服"""
    user_name = state.get("user_name", "用户")
    
    # 生成对话摘要供人工客服参考
    summary_response = llm_fast.invoke([
        SystemMessage("用3句话总结这段客服对话的核心问题，供人工客服快速了解情况。"),
        *state["messages"],
    ])
    
    reply = f"""好的，{user_name}，我现在为您转接人工客服。

**等待时间**：预计 3-5 分钟

**对话摘要（已发送给客服）**：
{summary_response.content}

转接期间请不要关闭窗口，客服接入后会主动打招呼。
如等待时间过长，您也可以拨打 **400-xxx-xxxx**。"""
    
    return {
        "messages": [AIMessage(reply)],
        "escalate_to_human": True,
    }


# ── 节点 6：满意度评估 ────────────────────────────────────

def ask_satisfaction(state: CustomerServiceState) -> dict:
    """在适当时机询问满意度"""
    reply = """很高兴能帮到您！

请问这次服务体验如何？（1-5分，5分最好）
您的反馈帮助我们持续改进服务 🙏"""
    
    return {"messages": [AIMessage(reply)]}
```

## 阶段四：图构建（graph.py）

```python
# graph.py
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
from state import CustomerServiceState
from nodes import (
    classify_intent, handle_faq, handle_ticket,
    handle_chat, escalate_to_human, ask_satisfaction,
)


def route_by_intent(state: CustomerServiceState) -> str:
    """根据意图路由到对应节点"""
    intent = state.get("intent", "unknown")
    
    if state.get("escalate_to_human"):
        return END
    
    routing = {
        "faq": "faq_handler",
        "ticket": "ticket_handler",
        "human": "human_escalation",
        "chat": "chat_handler",
        "unknown": "chat_handler",  # 未知意图走闲聊
    }
    return routing.get(intent, "chat_handler")


def should_ask_satisfaction(state: CustomerServiceState) -> str:
    """判断是否询问满意度"""
    # 转人工不询问
    if state.get("escalate_to_human"):
        return END
    
    # 每3轮或完成明确任务后询问
    turn_count = state.get("turn_count", 0)
    has_ticket = state.get("ticket_id") is not None
    
    if has_ticket or turn_count % 3 == 0:
        return "satisfaction"
    
    return END


def build_graph():
    graph = StateGraph(CustomerServiceState)
    
    # 添加节点
    graph.add_node("intent_classifier", classify_intent)
    graph.add_node("faq_handler", handle_faq)
    graph.add_node("ticket_handler", handle_ticket)
    graph.add_node("human_escalation", escalate_to_human)
    graph.add_node("chat_handler", handle_chat)
    graph.add_node("satisfaction", ask_satisfaction)
    
    # 入口
    graph.set_entry_point("intent_classifier")
    
    # 意图分类后路由
    graph.add_conditional_edges(
        "intent_classifier",
        route_by_intent,
        {
            "faq_handler": "faq_handler",
            "ticket_handler": "ticket_handler",
            "human_escalation": "human_escalation",
            "chat_handler": "chat_handler",
        }
    )
    
    # 各处理节点后决定是否询问满意度
    for node in ["faq_handler", "ticket_handler", "chat_handler"]:
        graph.add_conditional_edges(
            node,
            should_ask_satisfaction,
            {"satisfaction": "satisfaction", END: END}
        )
    
    graph.add_edge("human_escalation", END)
    graph.add_edge("satisfaction", END)
    
    # 带持久化的编译
    checkpointer = MemorySaver()
    return graph.compile(checkpointer=checkpointer)


service_graph = build_graph()
```

## 阶段五：WebSocket 服务（main.py）

```python
# main.py
from fastapi import FastAPI, WebSocket, WebSocketDisconnect
from langchain_core.messages import HumanMessage
from graph import service_graph
import json

app = FastAPI()


@app.websocket("/ws/{session_id}")
async def websocket_endpoint(websocket: WebSocket, session_id: str):
    await websocket.accept()
    
    config = {"configurable": {"thread_id": session_id}}
    
    # 初始化状态
    initial_state = {
        "messages": [],
        "intent": None,
        "user_id": session_id,
        "is_authenticated": False,
        "turn_count": 0,
        "escalate_to_human": False,
    }
    
    # 发送欢迎消息
    await websocket.send_json({
        "type": "welcome",
        "content": "你好！我是智能客服助手，有什么可以帮您的？",
    })
    
    try:
        while True:
            # 接收用户消息
            data = await websocket.receive_text()
            user_input = json.loads(data).get("message", "")
            
            if not user_input.strip():
                continue
            
            # 发送"正在输入"状态
            await websocket.send_json({"type": "typing", "content": True})
            
            # 运行图，流式收集事件
            async for event in service_graph.astream(
                {"messages": [HumanMessage(user_input)]},
                config=config,
            ):
                for node_name, node_output in event.items():
                    if node_name == "__end__":
                        continue
                    
                    messages = node_output.get("messages", [])
                    for msg in messages:
                        if hasattr(msg, "content"):
                            # 流式发送 AI 回复
                            await websocket.send_json({
                                "type": "message",
                                "content": msg.content,
                                "node": node_name,
                            })
                    
                    # 发送工单信息
                    if "ticket_id" in node_output:
                        await websocket.send_json({
                            "type": "ticket_created",
                            "ticket_id": node_output["ticket_id"],
                        })
                    
                    # 转人工通知
                    if node_output.get("escalate_to_human"):
                        await websocket.send_json({
                            "type": "escalate",
                            "content": "正在为您转接人工客服...",
                        })
            
            await websocket.send_json({"type": "typing", "content": False})
    
    except WebSocketDisconnect:
        print(f"会话 {session_id} 断开")


if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

import asyncio
import websockets
import json

async def simulate_conversation():
    uri = "ws://localhost:8000/ws/test_user_001"

                    print(f"[系统] {response['content']}")
                elif response["type"] == "typing" and not response["content"]:

## 关键设计决策

| 决策点 | 选择 | 原因 |
|--------|------|------|
| 路由策略 | LLM 意图分类 | 灵活，自然语言理解能力强 |
| 状态管理 | LangGraph StateGraph | 支持复杂分支、持久化、人工介入 |
| LLM 选择 | GPT-4o（对话）+ GPT-4o-mini（分类/提取） | 性价比优化，分类不需要强模型 |
| 会话持久化 | MemorySaver（开发）/ PostgresSaver（生产） | 开发简单，生产可靠 |
| 实时通信 | WebSocket + SSE | 双向实时通信，支持流式输出 |
